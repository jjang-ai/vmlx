Here is the Python code to reverse a singly linked list.

### The Solution

```python
class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

def reverse_linked_list(head):
    prev = None
    current = head
    
    while current is not None:
        next_temp = current.next  # Store the next node
        current.next = prev       # Reverse the link
        prev = current            # Move prev to current
        current = next_temp       # Move current to next node
        
    return prev  # prev is now the new head of the list

def print_list(head):
    """Helper function to print the list for verification."""
    current = head
    while current is not None:
        print(current.data, end=" -> ")
        current = current.next
    print("None")

if __name__ == "__main__":
    # Create a sample list: 1 -> 2 -> 3 -> 4 -> 5
    head = Node(1)
    head.next = Node(2)
    head.next.next = Node(3)
    head.next.next.next = Node(4)
    head.next.next.next = Node(5)

    print("Original List:")
    print_list(head)

    # Reverse the list
    new_head = reverse_linked_list(head)

    print("\nReversed List:")
    print_list(new_head)
```
