class Node:
    """Node class for singly linked list."""
    def __init__(self, value):
        self.value = value
        self.next = None

def reverse_linked_list(head):
    """
    Reverses a singly linked list iteratively.

    Args:
        head (Node): The head node of the linked list.

    Returns:
        Node: The new head of the reversed linked list.
    """
    prev = None
    current = head

    while current:
        next_node = current.next  # Save the next node
        current.next = prev       # Reverse the link
        prev = current            # Move prev forward
        current = next_node       # Move current forward

    return prev

def create_linked_list(values):
    """
    Creates a linked list from a list of values.

    Args:
        values (list): List of values to create nodes from.

    Returns:
        Node: The head of the created linked list.
    """
    if not values:
        return None

    head = Node(values[0])
    current = head

    for value in values[1:]:
        current.next = Node(value)
        current = current.next

    return head

def print_linked_list(head):
    """
    Prints the values of a linked list.

    Args:
        head (Node): The head node of the linked list.
    """
    values = []
    current = head
    while current:
        values.append(str(current.value))
        current = current.next
    print(" -> ".join(values))
