```python
class Node:
    """A simple node class for a singly linked list."""
    def __init__(self, data):
        self.data = data
        self.next = None

def reverse_linked_list(head):
    """
    Reverse a singly linked list.
    
    Args:
        head: The head node of the linked list
        
    Returns:
        The new head of the reversed linked list
    """
    prev = None
    current = head
    
    while current is not None:
        next_node = current.next  # Store the next node
        current.next = prev       # Reverse the link
        prev = current            # Move prev one step forward
        current = next_node       # Move current one step forward
    
    return prev  # prev is now the new head

# Example in __main__
if __name__ == "__main__":
    # Create a linked list: 1 -> 2 -> 3 -> 4 -> 5
    node1 = Node(1)
    node2 = Node(2)
    node3 = Node(3)
    node4 = Node(4)
    node5 = Node(5)
    
    node1.next = node2
    node2.next = node3
    node3.next = node4
    node4.next = node5
    
    # Print original list
    print("Original list:", end=" ")
    current = node1
    while current:
        print(current.data, end=" -> " if current.next else "\n")
    
    # Reverse the list
    reversed_head = reverse_linked_list(node1)
    
    # Print reversed list
    print("Reversed list:", end=" ")
    current = reversed_head
    while current:
        print(current.data, end=" -> " if current.next else "\n")
```

**Output:**
```
Original list: 1 -> 2 -> 3 -> 4 -> 5
Reversed list: 5 -> 4 -> 3 -> 2 -> 1
```

This solution uses an iterative approach with O(1) space complexity and O(n) time complexity. It reverses the list by iteratively changing the `next` pointers of each node to point to the previous node, and finally returns the new head (which was originally the last node).
