class Node:
    """A singly linked list node."""
    
    def __init__(self, value: int, next: 'Node | None' = None):
        self.value = value
        self.next = next
    
    def __repr__(self):
        return f"Node({self.value})"


def reverse_linked_list(head: Node | None) -> Node | None:
    """
    Reverse a singly linked list iteratively.
    
    Args:
        head: The head node of the linked list (or None for empty list)
    
    Returns:
        The new head node of the reversed linked list
    
    Time:  O(n) - single pass through the list
    Space: O(1) - only using constant extra space
    """
    prev = None
    current = head
    
    while current:
        # Store the next node before we break the link
        next_temp = current.next
        
        # Reverse the pointer
        current.next = prev
        
        # Move pointers forward
        prev = current
        current = next_temp
    
    # prev is now pointing to the new head
    return prev


def print_list(head: Node | None) -> None:
    """Helper function to print a linked list."""
    values = []
    current = head
    while current:
        values.append(str(current.value))
        current = current.next
    print(" -> ".join(values) or "(empty)")


if __name__ == "__main__":
    # Create a linked list: 1 -> 2 -> 3 -> 4 -> 5
    head = Node(1, Node(2, Node(3, Node(4, Node(5)))))
    
    print("Original list:")
    print_list(head)
    
    # Reverse the list
    new_head = reverse_linked_list(head)
    
    print("\nReversed list:")
    print_list(new_head)
