```python
class Node:
    """A class to represent a node in a singly linked list."""
    def __init__(self, data):
        self.data = data
        self.next = None

def reverse_linked_list(head):
    """
    Reverses a singly linked list.
    
    Args:
        head: The head node of the linked list.
    
    Returns:
        The new head of the reversed linked list.
    """
    prev = None
    current = head
    
    while current is not None:
        next_node = current.next  # Store the next node
        current.next = prev       # Reverse the link
        prev = current            # Move prev one step forward
        current = next_node       # Move current one step forward
    
    return prev  # prev is now the new head

# Example usage in __main__
if __name__ == "__main__":
    # Create a linked list: 1 -> 2 -> 3 -> 4 -> 5
    # Step-by-step creation
    node1 = Node(1)
    node2 = Node(2)
    node3 = Node(3)
    node4 = Node(4)
    node5 = Node(5)
    
    node1.next = node2
    node2.next = node3
    node3.next = node4
    node4.next = node5
    
    # Print original list
    def print_list(head):
        result = []
        current = head
        while current:
            result.append(current.data)
            current = current.next
        return result
    
    print("Original list:", print_list(node1))
    
    # Reverse the list
    reversed_head = reverse_linked_list(node1)
    
    # Print reversed list
    print("Reversed list:", print_list(reversed_head))
```

**Output:**
```
Original list: [1, 2, 3, 4, 5]
Reversed list: [5, 4, 3, 2, 1]
```

This function reverses a singly linked list in O(n) time and O(1) space using an iterative approach. It iterates through the list, reversing the direction of each node's `next` pointer as it goes, and returns the new head of the reversed list.
